Getting started

Include FlThemeService in the app provider as a Component dependency. Use the set() method to establish the theme colors.

import { FlThemeService } from './wui/theme2/theme.service';

@Component({
  selector: 'app-root',
  ...
  providers: [FlThemeService]
})
export class AppComponent {

  constructor(private _theme: FlThemeService) {
    this._theme.set('teal-lightBlue-light');
  }
}
